917  ilength--; /* don't count the - sign as a sig fig. */
918  }
919  if (isFraction)
920  prec = flength;
921  else if (flags & STR_FLOAT)
922  {
923  prec = MAXFLOATDIGITS - ilength - 1;
924  }
925  else
926  {
927  prec = MAXDOUBLEDIGITS - ilength - 1;
928  }
929  if (ilength == 1 && ipart == 0)
930  {
931  prec++; // 0 on left of 0.1 isn't a sig fig, one more avail for right
932  }
933  if (prec >= flength)
934  {
935  prec = flength - 1; // just in case we screwed up some how
936  }
937  for (; prec >= 0; prec--)
938  {
939  if (fstring.at(prec) != '0')
940  break;
941  }
942  prec++;
943  // At this point prec is number of sig figs to the right of the decimal
944 
945  // Step 3 ////////////////////////////////////////////////////////////////////
946 
947  /* Find the max number of digits to the right of the decimal we can allow,
948  based on a total number of <length> digits including the '-' and the '.'
949  This is necessary because we might get round off errors above causing
950  garbage values towards the end of the string causing prec to be more
951  than it should be. Here's an example if <length> is 15. */
952 
953  /* 123456789012345 123456789012345 maxdigits */
954  /* fabs(value) >= fabs(-100000000000.0) 1 */
955  /* fabs(-100000000000.0) > fabs(value) >= fabs(-10000000000.00) 2 */
956  /* fabs(-10000000000.00) > fabs(value) >= fabs(-1000000000.000) 3 */
957  /* fabs(-1000000000.000) > fabs(value) >= fabs(-100000000.0000) 4 */
958  /* fabs(-100000000.0000) > fabs(value) >= fabs(-10000000.00000) 5 */
959  /* fabs(-10000000.00000) > fabs(value) >= fabs(-1000000.000000) 6 */
960  /* fabs(-1000000.000000) > fabs(value) >= fabs(-100000.0000000) 7 */
961  /* fabs(-100000.0000000) > fabs(value) >= fabs(-10000.00000000) 8 */
962  /* fabs(-10000.00000000) > fabs(value) >= fabs(-1000.000000000) 9 */
963  /* fabs(-1000.000000000) > fabs(value) >= fabs(-100.0000000000) 10 */
964  /* fabs(-100.0000000000) > fabs(value) >= fabs(-10.00000000000) 11 */
965  /* fabs(-10.00000000000) > fabs(value) >= fabs(-1.000000000000) 12 */
966  /* fabs(-1.000000000000) > fabs(value) >= fabs(-.1000000000000) 12 */
967  /* fabs(-.1000000000000) > fabs(value) >= fabs(-.0100000000000) 12 */
968 
969  if (flags & STR_SCIENTIFIC)
970  {
971  maxdigits = (short)(length - 8); // 8 places used for -1. and e+012
972  }
973  else
974  {
975  maxdigits = (short)(length - 3); // 3 because 1 for + or -, 1 for 0, 1 for . (-0.)
976  double testval;
977  testval = 10.0;
978  if (!LT_EPS(fabs(value), testval, DBL_EPSILON))
979  {
980  do
981  {
982  maxdigits--;
983  testval *= 10.0;
984  } while (!LT_EPS(fabs(value), testval, DBL_EPSILON));
985  }
986  }
987  if (!(value < 0.0))
988  {
989  maxdigits++; // If value is positive, you get one more
990  }
991 
992  // Step 4 ////////////////////////////////////////////////////////////////////
993 
994  // Compare the maximum number of significant digits to the right of the
995  // decimal allowed <maxdigits> with how many we have <prec>. If we have
996  // more than we're allowed, we have to throw out the extra. If we have
997  // to throw some out, we have to see if the ones we have left are
998  // still significant - if the ones on the right are 0, then they're no
999  // longer significant.
1000 
1001  if (prec > maxdigits)
1002  {
1003  prec = maxdigits;
1004  // format = (boost::format("%s%d.%d%s") % "%" % ilength % prec % "f").str();
1005  // fstring = (boost::format(format) % value).str();
1006  // charptr = fstring.find('.');
1007  // if (charptr != std::string::npos) {
1008  // charptr++;
1009  // fstring = fstring.substr(charptr);
1010  //}
1011  if ((int)fstring.size() > prec && fstring.at(prec - 1) == '0')
1012  {
1013  std::stringstream ss;
1014  ss << fstring.at(prec);
1015  int val;
1016  ss >> val;
1017  if (val > 4)
1018  fstring.at(prec - 1) = '1';
1019  }
1020  for (; prec >= 1; prec--)
1021  {
1022  if (fstring.at(prec - 1) != '0')
1023  {
1024  break;
1025  }
1026  }
1027  }
1028  if (prec < 1)
1029  {
1030  prec = 1; // always show 5.0 instead of just 5 to indicate it's a float
1031  }
1032 
1033  return prec;
1034 } // Prec
1035

1074 std::string STRstd(double a_value, int a_n /*=-1*/, int width /*=15*/, int flags /*=0*/)
1075 {
1076  std::string str;
1077  int prec = 0;
1078 
1079  // put this check in for old stuff
1080  if (a_n >= 800)
1081  {
1082  XM_ASSERT(false);
1083  a_n = -1;
1084  }
1085 
1086  // check for invalid values
1087 #if BOOST_OS_WINDOWS
1088  if (!_finite(a_value))
1089  {
1090  switch (_fpclass(a_value))
1091  {
1092  case _FPCLASS_NINF:
1093  str = "-INF";
1094  break;
1095  case _FPCLASS_PINF:
1096  str = "+INF";
1097  break;
1098  default:
1099  str = "NaN";
1100  break;
1101  }
1102  return str;
1103  }
1104 #else
1105  if (!std::isfinite(a_value))
1106  {
1107  switch (std::fpclassify(a_value))
1108  {
1109  case FP_INFINITE:
1110  if (std::signbit(a_value))
1111  str = "-INF";
1112  else
1113  str = "+INF";
1114  break;
1115  default:
1116  str = "NaN";
1117  break;
1118  }
1119  return str;
1120  }
1121 #endif
1122 
1123  // if not specifying an exact prec, autocompute
1124  if ((a_n == -1) || (flags & STR_USEMAXPREC))
1125  {
1126  // if using max precision and the number is small, truncate the number to
1127  // max precision here before Prec() switches the number to scientific
1128  // notation (a number that is beyond the limited precision range)
1129  if ((flags & STR_USEMAXPREC) && LT_TOL(fabs(a_value), 1e-4, DBL_EPSILON))
1130  {
1131  a_value *= pow(10.0, a_n);
1132  a_value = floor(a_value + 0.5);
1133  a_value /= pow(10.0, a_n);
1134  }
1135  // figure out the auto-computed prec is
1136  try
1137  {
1138  prec = xus::Prec(a_value, flags, width);
1139  }
1140  catch (std::exception&)
1141  {
1142  // we had an error. Generally with casting the modf integer part
1143  return "";
1144  }
1145  if ((flags & STR_USEMAXPREC) && prec > a_n)
1146  {
1147  prec = a_n;
1148  }
1149  }
1150  else
1151  {
1152  prec = a_n;
1153  }
1154 
1155  // Format the format string, then use it to format str
1156 
1157  std::string format;
1158  if (flags & STR_SCIENTIFIC)
1159  {
1160  // sprintf(format, "%%.%de", Miabs(prec));
1161  format = (boost::format("%%.%de") % Miabs(prec)).str();
1162  }
1163  else
1164  {
1165  // sprintf(format, "%%.%dlf", prec);
1166  // I added Miabs because according to my book it must be nonnegative
1167  // and some combinations result in a negative prec, which crashes. -MJK
1168  format = (boost::format("%%.%dlf") % Miabs(prec)).str();
1169  }
1170 
1171  if (!(flags & STR_WITHCOMMAS))
1172  {
1173  // str.Format(format, a_value);
1174  str = (boost::format(format) % a_value).str();
1175  }
1176  else
1177  {
1178  // Use the system locale to get the commas (or whatever else)
1179  std::locale loc(""); // system locale
1180  str = (boost::format(format, loc) % a_value).str();
1181  }
1182 
1183  // in some cases when specifying max prec, trailing zeros can occur. For
1184  // example, if the number is 5.0003 and the maxprec is 3, the string right
1185  // here will be 5.000 instead of 5.0.
1186  if ((a_n == -1) || (flags & STR_USEMAXPREC))
1187  {
1188  // see if there's a decimal point
1189  size_t index = str.find('.');
1190  if (index != std::string::npos)
1191  {
1192  if ((index = str.find_first_of("eE")) != std::string::npos)
1193  {
1194  // Scientific. Start at 'e' or 'E' and go left, removing zeros
1195  index--;
1196  while (index > 0 && str.at(index) == '0' && str.at(index - 1) != '.')
1197  {
1198  str.erase(index, 1);
1199  index--;
1200  }
1201  }
1202  else
1203  {
1204  // take off any trailing zeros
1205  xus::trim_right(str, "0");
1206  // make sure there's at least one zero
1207  if (str[str.size() - 1] == '.')
1208  {
1209  str += '0';
1210  }
1211  }
1212  }
1213  }
1214 
1215  // in some cases you want the string to always be a certain number of
1216  // character. This is most common with FORTRASH fixed format garbage.
1217  if (flags & STR_FULLWIDTH)
1218  {
1219  size_t diff = width - str.size();
1220  if (diff > 0)
1221  {
1222  std::string str1, str2;
1223  for (size_t i = 0; i < diff; i++)
1224  str1 += " ";
1225  str2 = str;
1226  str = str1 + str2;
1227  }
1228  }
1229 
1230  // if the string is "-0.0" remove the negative sign
1231  if (str == "-0.0")
1232  {
1233  str = "0.0";
1234  }
1235 
1236  return str;
1237 } // STRstd
